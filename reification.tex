\begin{frame}[fragile]{An example}
  \begin{itemize}
  \item \structure{Logic program} \texttt{ezy.lp}
\begin{lstlisting}
{a}.
b :- a.
c :- not a.
\end{lstlisting}
  \item<2-> \structure{Implicit statements}
\begin{lstlisting}
#show a.
#show b.
#show c.
\end{lstlisting}
  \item<3-> \structure{Reification}
\begin{lstlisting}
$ clingo --output=reify ezy.lp
\end{lstlisting}
  \end{itemize}
\end{frame}
% ----------------------------------------------------------------------
